\subsection{Derechos patrimoniales}
\frame
{
	\frametitle{Derechos patrimoniales}
	\begin{itemize}
	\item Disponer de una obra a titulo gratuito
	\item Aprovecharla, con fines de lucro o sin él, a través de cualquier medio de reproducción, multiplicación, o difusión conocido o por conocer
	\item La edición, o cualquier otra forma de reproducción
	\end{itemize}
}

\subsection{Derechos morales}
\frame
{
	\frametitle{Derechos morales}
	\begin{itemize}
	\item Reivindicar la paternidad de su obra
	\item Oponerse a toda deformación, mutilación u otra modificación de la obra, cuando tales actos puedan causar a su reputación 
	\item Conservar su obra inédita o anónima hasta su fallecimiento, o después de él 
	\item Modificar, antes o después de su publicación
	\item Retirar de la circulación o suspender cualquier forma de utilización aunque ella hubiese sido previamente autorizada.
	\end{itemize}
}
